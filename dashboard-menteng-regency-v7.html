// ─── MENTENG REGENCY — Apps Script v4 ───
// Support JSONP (callback) + direct JSON — bekerja dari file:// maupun https://

function doGet(e) {
  const action   = e.parameter.action   || 'read';
  const callback = e.parameter.callback || '';   // JSONP support

  let result;

  if (action === 'write') {
    try {
      const ss        = SpreadsheetApp.getActiveSpreadsheet();
      const sheetName = e.parameter.sheet;
      const headers   = JSON.parse(e.parameter.headers);
      const row       = JSON.parse(e.parameter.row);
      let sheet = ss.getSheetByName(sheetName);
      if (!sheet) sheet = ss.insertSheet(sheetName);
      if (sheet.getLastRow() === 0) sheet.appendRow(headers);
      sheet.appendRow(row);
      result = { status: 'ok', sheet: sheetName, rows: sheet.getLastRow() };
    } catch(err) {
      result = { status: 'error', error: err.toString() };
    }
  } else {
    try {
      const ss        = SpreadsheetApp.getActiveSpreadsheet();
      const sheetName = e.parameter.sheet;
      if (!sheetName) { result = { rows: [] }; }
      else {
        const sheet = ss.getSheetByName(sheetName);
        result = sheet ? { rows: sheet.getDataRange().getValues() } : { rows: [] };
      }
    } catch(err) {
      result = { rows: [], error: err.toString() };
    }
  }

  const json = JSON.stringify(result);
  // Jika ada callback → JSONP (untuk file:// browser)
  if (callback) {
    return ContentService
      .createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  // Biasa → JSON
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  // Redirect ke doGet handler
  return doGet(e);
}

📋 Kode Apps Script v4
Copy semua kode ini → paste di Google Apps Script → Deploy sebagai Web App
// ═══════════════════════════════════════════════
// MENTENG REGENCY — Apps Script v5 FINAL
// Deploy sebagai Web App:
//   Execute as: Me
//   Who has access: Anyone
// ═══════════════════════════════════════════════

function doGet(e) {
  const action   = (e && e.parameter && e.parameter.action) || 'read';
  const callback = (e && e.parameter && e.parameter.callback) || '';

  let result;

  // ── WRITE ──────────────────────────────────────
  if (action === 'write') {
    try {
      const ss        = SpreadsheetApp.getActiveSpreadsheet();
      const sheetName = e.parameter.sheet;
      const headers   = JSON.parse(decodeURIComponent(e.parameter.headers));
      const row       = JSON.parse(decodeURIComponent(e.parameter.row));

      let sheet = ss.getSheetByName(sheetName);
      if (!sheet) sheet = ss.insertSheet(sheetName);
      if (sheet.getLastRow() === 0) sheet.appendRow(headers);
      sheet.appendRow(row);
      // Force Bulan column as plain text — prevent date auto-conversion
      const bulanIdx = headers.indexOf('Bulan');
      if(bulanIdx >= 0) {
        sheet.getRange(sheet.getLastRow(), bulanIdx + 1).setNumberFormat('@STRING@');
      }
      result = { status: 'ok', sheet: sheetName, totalRows: sheet.getLastRow() - 1 };
    } catch(err) {
      result = { status: 'error', error: err.toString() };
    }

  // ── READ ───────────────────────────────────────
  } else if (action === 'read') {
    try {
      const ss        = SpreadsheetApp.getActiveSpreadsheet();
      const sheetName = e.parameter.sheet;
      if (!sheetName) {
        result = { rows: [] };
      } else {
        const sheet = ss.getSheetByName(sheetName);
        result = sheet ? { rows: sheet.getDataRange().getValues() } : { rows: [] };
      }
    } catch(err) {
      result = { rows: [], error: err.toString() };
    }

  // ── PING (test koneksi) ────────────────────────
  } else if (action === 'ping') {
    result = { status: 'ok', message: 'Menteng Regency Apps Script aktif', time: new Date().toISOString() };

  } else {
    result = { status: 'error', error: 'Unknown action: ' + action };
  }

  const json = JSON.stringify(result);

  // JSONP untuk file:// browser
  if (callback) {
    return ContentService
      .createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

// doPost redirect ke doGet (fallback)
function doPost(e) {
  return doGet(e);
}
